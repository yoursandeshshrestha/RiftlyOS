# Agency OS — Plan 01: Foundation Implementation Plan

> **React + Supabase amendment (v2.1):** This plan was written for a greenfield **Next.js + Drizzle** app. We are **evolving `riftly-dashboard`** with **React (Vite) + Supabase** instead. **Keep all steps below** for requirements, RLS policies, and acceptance criteria; **map file paths and server patterns** using [Plan 00 §2.1](./2026-06-28-00-conventions-and-build-order.md#21-react--supabase-implementation-chosen--v21). **Skip** Task 1 (`create next-app`). Start from the existing repo: Supabase migrations, `AuthContext`/`WorkspaceContext`, `src/App.tsx` routing.

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Stand up the Agency OS codebase with a multi-tenant Postgres database whose tenant isolation is enforced by Postgres Row-Level Security and proven by tests, plus authentication, role-based routing, and an app shell — the foundation every later plan builds on.

**Architecture:** Next.js 16 (App Router, TypeScript strict) on Supabase Postgres. Tenancy model: `organizations` are tenants (one `agency` org + one `client` org per client); `memberships` map users → org → role (`founder | team | client`). Isolation is enforced at the database by RLS policies that call SECURITY DEFINER helper functions; agency staff get cross-client read, clients are scoped to their own org. A Supabase custom-access-token hook stamps `app_role` + `org_id` into the JWT for the app layer. Defense in depth: the app also scopes every query.

**Tech Stack:** Next.js 16 · TypeScript · pnpm · Supabase (Postgres + Auth, local CLI) · Drizzle ORM + drizzle-kit · postgres.js · Tailwind CSS + shadcn/ui · Vitest (unit/integration incl. RLS tests) · GitHub Actions CI.

**Prerequisites the developer needs installed:** Node 20+, pnpm 9+, Docker (for `supabase start`), and the Supabase CLI (`pnpm dlx supabase` works without global install).

```ts
import { pgTable, pgEnum, uuid, text, timestamp, unique } from 'drizzle-orm/pg-core'

export const orgType = pgEnum('org_type', ['agency', 'client'])
export const appRole = pgEnum('app_role', ['founder', 'team', 'client'])

export const organizations = pgTable('organizations', {
  id: uuid('id').defaultRandom().primaryKey(),
  type: orgType('type').notNull(),
  name: text('name').notNull(),
  slug: text('slug').notNull().unique(),
  createdAt: timestamp('created_at', { withTimezone: true }).defaultNow().notNull(),
})

// Mirrors auth.users (id == auth.users.id). The FK is added in the RLS SQL migration
// (Task 6) because Drizzle does not manage the auth schema.
export const profiles = pgTable('profiles', {
  id: uuid('id').primaryKey(),
  email: text('email').notNull(),
  fullName: text('full_name'),
  createdAt: timestamp('created_at', { withTimezone: true }).defaultNow().notNull(),
})

export const memberships = pgTable(
  'memberships',
  {
    id: uuid('id').defaultRandom().primaryKey(),
    userId: uuid('user_id')
      .notNull()
      .references(() => profiles.id, { onDelete: 'cascade' }),
    organizationId: uuid('organization_id')
      .notNull()
      .references(() => organizations.id, { onDelete: 'cascade' }),
    role: appRole('role').notNull(),
    createdAt: timestamp('created_at', { withTimezone: true }).defaultNow().notNull(),
  },
  (t) => ({ uniqUserOrg: unique('uniq_user_org').on(t.userId, t.organizationId) }),
)
```

- [ ] **Step 1: Write `scripts/seed.ts`**

```ts
import 'dotenv/config'
import { createClient } from '@supabase/supabase-js'
import { drizzle } from 'drizzle-orm/postgres-js'
import postgres from 'postgres'
import { eq } from 'drizzle-orm'
import * as schema from '../src/db/schema'

const admin = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!,
  { auth: { autoRefreshToken: false, persistSession: false } },
)
const db = drizzle(postgres(process.env.DATABASE_URL!, { prepare: false }), { schema })

async function ensureUser(email: string, fullName: string): Promise<string> {
  // Idempotent: reuse if the user already exists.
  const existing = await db.select().from(schema.profiles).where(eq(schema.profiles.email, email))
  if (existing[0]) return existing[0].id

  const { data, error } = await admin.auth.admin.createUser({
    email,
    password: 'Password123!',
    email_confirm: true,
    user_metadata: { full_name: fullName },
  })
  if (error || !data.user) throw error ?? new Error('createUser failed')
  await db.insert(schema.profiles).values({ id: data.user.id, email, fullName })
  return data.user.id
}

async function ensureOrg(slug: string, name: string, type: 'agency' | 'client'): Promise<string> {
  const existing = await db.select().from(schema.organizations).where(eq(schema.organizations.slug, slug))
  if (existing[0]) return existing[0].id
  const [row] = await db.insert(schema.organizations).values({ slug, name, type }).returning()
  return row!.id
}

async function ensureMembership(userId: string, organizationId: string, role: 'founder' | 'team' | 'client') {
  await db
    .insert(schema.memberships)
    .values({ userId, organizationId, role })
    .onConflictDoNothing({ target: [schema.memberships.userId, schema.memberships.organizationId] })
}

async function main() {
  const agency = await ensureOrg('milktree', 'Milktree Agency', 'agency')
  const client1 = await ensureOrg('client-one', 'Client One Ltd', 'client')
  const client2 = await ensureOrg('client-two', 'Client Two Ltd', 'client')

  const founder = await ensureUser('founder@milktreeagency.com', 'Levi Eweka')
  const u1 = await ensureUser('user1@clientone.com', 'Client One User')
  const u2 = await ensureUser('user2@clienttwo.com', 'Client Two User')

  await ensureMembership(founder, agency, 'founder')
  await ensureMembership(u1, client1, 'client')
  await ensureMembership(u2, client2, 'client')

  console.log(JSON.stringify({ agency, client1, client2, founder, u1, u2 }, null, 2))
  process.exit(0)
}

main().catch((e) => {
  console.error(e)
  process.exit(1)
})
```

```ts
import 'dotenv/config'
import postgres from 'postgres'

// Single connection; tests run serially within transactions.
export const sql = postgres(process.env.DATABASE_URL!, { max: 1, prepare: false })

// Run `fn` as the given auth user: assume the non-privileged `authenticated`
// role (so RLS applies) and set the JWT claim `sub` that auth.uid() reads.
export async function asUser<T>(userId: string, fn: (tx: postgres.TransactionSql) => Promise<T>): Promise<T> {
  return sql.begin(async (tx) => {
    await tx`set local role authenticated`
    const claims = JSON.stringify({ sub: userId, role: 'authenticated' })
    await tx`select set_config('request.jwt.claims', ${claims}, true)`
    return fn(tx)
  })
}

export async function userIdByEmail(email: string): Promise<string> {
  const rows = await sql`select id from public.profiles where email = ${email}`
  if (!rows[0]) throw new Error(`seed user not found: ${email} (run pnpm db:seed)`)
  return rows[0].id as string
}
```

```sql
-- Link profiles to auth.users (Drizzle can't reference the auth schema).
alter table public.profiles
  add constraint profiles_id_fk_auth_users
  foreign key (id) references auth.users (id) on delete cascade;

-- Performance: RLS policies filter by organization_id; index it leading.
create index if not exists idx_memberships_user on public.memberships (user_id, organization_id);
create index if not exists idx_memberships_org on public.memberships (organization_id, user_id);

-- Helper: is the current user agency staff (founder/team in an agency-type org)?
create or replace function public.is_agency_staff()
returns boolean
language sql
security definer
stable
set search_path = public
as $$
  select exists (
    select 1
    from public.memberships m
    join public.organizations o on o.id = m.organization_id
    where m.user_id = auth.uid()
      and o.type = 'agency'
      and m.role in ('founder', 'team')
  );
$$;

-- Helper: can the current user access rows belonging to target_org?
create or replace function public.has_org_access(target_org uuid)
returns boolean
language sql
security definer
stable
set search_path = public
as $$
  select public.is_agency_staff()
      or exists (
        select 1 from public.memberships m
        where m.user_id = auth.uid() and m.organization_id = target_org
      );
$$;

-- Enable RLS.
alter table public.organizations enable row level security;
alter table public.memberships  enable row level security;
alter table public.profiles     enable row level security;

-- organizations: visible if the user has access to that org.
create policy org_select on public.organizations
  for select using (public.has_org_access(id));

-- memberships: a user sees their own rows; staff see all.
create policy memb_select on public.memberships
  for select using (user_id = auth.uid() or public.is_agency_staff());

-- profiles: a user sees their own; staff see all.
create policy profile_select on public.profiles
  for select using (id = auth.uid() or public.is_agency_staff());
```

- [ ] **Step 3: Write the access-token hook migration `drizzle/0002_access_token_hook.sql`**

Create via `pnpm db:generate --custom --name=access_token_hook`, then fill it:

```sql
-- Stamp app_role + org_id into the JWT so the app can route without a DB round-trip.
-- For staff with multiple memberships, prefer the agency/founder role.
create or replace function public.custom_access_token_hook(event jsonb)
returns jsonb
language plpgsql
stable
as $$
declare
  claims jsonb;
  v_user uuid := (event ->> 'user_id')::uuid;
  v_role text;
  v_org uuid;
begin
  select m.role, m.organization_id
    into v_role, v_org
  from public.memberships m
  join public.organizations o on o.id = m.organization_id
  where m.user_id = v_user
  order by
    case when o.type = 'agency' then 0 else 1 end,
    case m.role when 'founder' then 0 when 'team' then 1 else 2 end
  limit 1;

  claims := coalesce(event -> 'claims', '{}'::jsonb);
  if v_role is not null then
    claims := jsonb_set(claims, '{app_role}', to_jsonb(v_role));
    claims := jsonb_set(claims, '{org_id}', to_jsonb(v_org));
  end if;

  return jsonb_set(event, '{claims}', claims);
end;
$$;

grant execute on function public.custom_access_token_hook(jsonb) to supabase_auth_admin;
grant usage on schema public to supabase_auth_admin;
```

- [ ] **Step 6: Write session/role helpers `src/lib/auth.ts`**

```ts
import { createSupabaseServerClient } from './supabase/server'
import type { AppRole } from '@/db/types'

export type SessionInfo = {
  userId: string
  email: string
  role: AppRole | null
  orgId: string | null
}

export async function getSession(): Promise<SessionInfo | null> {
  const supabase = await createSupabaseServerClient()
  const { data } = await supabase.auth.getUser()
  if (!data.user) return null

  // app_role/org_id are stamped into the JWT by the custom-access-token hook (Step 3).
  const accessToken = (await supabase.auth.getSession()).data.session?.access_token
  const payload = decodeJwt(accessToken)
  const role = (payload?.app_role as AppRole | undefined) ?? null
  const orgId = (payload?.org_id as string | undefined) ?? null

  return { userId: data.user.id, email: data.user.email ?? '', role, orgId }
}

export function isStaff(role: AppRole | null): boolean {
  return role === 'founder' || role === 'team'
}

function decodeJwt(token: string | undefined): Record<string, unknown> | null {
  if (!token) return null
  try {
    const payload = token.split('.')[1]
    return JSON.parse(Buffer.from(payload, 'base64url').toString('utf8'))
  } catch {
    return null
  }
}
```

- [ ] **Step 1: Middleware `src/middleware.ts` (refresh session + protect routes)**

```ts
import { NextResponse, type NextRequest } from 'next/server'
import { createServerClient } from '@supabase/ssr'

const PUBLIC_PATHS = ['/login', '/auth']

export async function middleware(request: NextRequest) {
  let response = NextResponse.next({ request })
  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll: () => request.cookies.getAll(),
        setAll: (toSet) => {
          toSet.forEach(({ name, value }) => request.cookies.set(name, value))
          response = NextResponse.next({ request })
          toSet.forEach(({ name, value, options }) => response.cookies.set(name, value, options))
        },
      },
    },
  )

  const { data } = await supabase.auth.getUser()
  const path = request.nextUrl.pathname
  const isPublic = PUBLIC_PATHS.some((p) => path.startsWith(p))

  if (!data.user && !isPublic) {
    return NextResponse.redirect(new URL('/login', request.url))
  }
  return response
}

export const config = {
  matcher: ['/((?!_next/static|_next/image|favicon.ico|.*\\.(?:svg|png|jpg|jpeg|gif|webp)$).*)'],
}
```

```tsx
'use client'
import { useState } from 'react'
import { useRouter } from 'next/navigation'
import { createSupabaseBrowserClient } from '@/lib/supabase/client'
import { Button } from '@/components/ui/button'

export default function LoginPage() {
  const router = useRouter()
  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')
  const [error, setError] = useState<string | null>(null)

  async function signIn(e: React.FormEvent) {
    e.preventDefault()
    setError(null)
    const supabase = createSupabaseBrowserClient()
    const { error } = await supabase.auth.signInWithPassword({ email, password })
    if (error) return setError(error.message)
    router.push('/')
    router.refresh()
  }

  return (
    <main className="mx-auto flex min-h-screen max-w-sm flex-col justify-center gap-4 p-6">
      <h1 className="text-xl font-semibold">Agency OS</h1>

          value={email} onChange={(e) => setEmail(e.target.value)} required />
        <input className="rounded border p-2" type="password" placeholder="Password"
          value={password} onChange={(e) => setPassword(e.target.value)} required />
        {error && <p className="text-sm text-red-600">{error}</p>}
        <Button type="submit">Sign in</Button>
```
